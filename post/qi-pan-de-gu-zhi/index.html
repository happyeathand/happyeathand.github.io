#include<stdio.h>

int main()
{
	int n;
	scanf("%d",&n);
	int a[n][n]={0};
	int i,j;
	int ili,lli,x,y;
	
	for(i=0;i<n;i++)
	{
		for(j=0;j<n;j++)
		{
			a[i][j]=0;
		}
	}
	
	if(n%2!=0)
	{
		x=n/2;
		y=n/2;
	}
	else
	{
		x=n/2-1;
		y=n/2-1;
	}
	for(i=1;i<=x;i++)
	{
		int flag=0;
		for(j=i;j<=y;j++)
		{
			if(flag==0)
			{
				for(int z=0;z<n;z++)
				{
					a[i][z]=a[i-1][z];
					a[n-i-1][z]=a[i][z];
					a[n-i-1][n-z-1]=a[i][z];
				}
				flag++;
			}
			a[i][j]+=1;
			a[n-i-1][j]=a[i][j];
			a[i][n-j-1]=a[i][j];
			a[n-i-1][n-j-1]=a[i][j];
		}
	}
	for(i=0;i<n;i++)
	{
		for(j=0;j<n;j++)
		{
			printf("%d ",a[i][j]);
			if(j==n-1) printf("\n");
		}
	}
	return 0;
}
